### Key Best Practices to Implement

1. **Price range filter (0.10-0.85):** Avoid near-certain outcomes (>0.85 = no margin) and ultra-long-shots (<0.10 = high loss rate)
2. **New positions only:** Only copy when trader opens a NEW position (not averaging into existing)
3. **Exit mirroring:** When tracked trader sells, we sell proportionally
4. **Multi-wallet confirmation:** Only enter when 2+ tracked wallets buy the same outcome
5. **Time-to-resolution filter:** Skip markets resolving in <2 hours (no time to profit) or >30 days (capital lockup)
6. **Position cap:** Max exposure per market ($25) and per session ($200/day)
7. **Min trigger amount:** Only copy trades where trader puts in >$50 (filters noise/test trades)
8. **Bankroll safety:** Never risk more than 5% of balance on one trade, stop trading at 20% drawdown

// src/executor.ts
import { execSync, ExecSyncOptionsWithStringEncoding } from "child_process";

const BULLPEN = process.env.BULLPEN_PATH || "bullpen";
const CMD_TIMEOUT = 30_000;

export interface BullpenResult {
  stdout: string;
  stderr: string;
  success: boolean;
  data: any | null;
  error: string | null;
}

export function parseBullpenOutput(raw: string): any | null {
  // Try array first
  const arrStart = raw.indexOf("[");
  const arrEnd = raw.lastIndexOf("]");
  if (arrStart !== -1 && arrEnd > arrStart) {
    try {
      return JSON.parse(raw.substring(arrStart, arrEnd + 1));
    } catch {}
  }
  // Try object
  const objStart = raw.indexOf("{");
  const objEnd = raw.lastIndexOf("}");
  if (objStart !== -1 && objEnd > objStart) {
    try {
      return JSON.parse(raw.substring(objStart, objEnd + 1));
    } catch {}
  }
  return null;
}

export function parseBullpenError(stderr: string): string {
  // Strip "Error: " prefix if present
  const match = stderr.match(/Error:\s*(.+)/i);
  return match ? match[1].trim() : stderr.trim();
}

export function bullpenExec(args: string): BullpenResult {
  try {
    // Use spawnSync-style to capture both stdout and stderr
    const stdout = execSync(`${BULLPEN} ${args} 2>/tmp/bullpen-stderr.txt`, {
      encoding: "utf-8",
      timeout: CMD_TIMEOUT,
    }).trim();

    let stderr = "";
    try {
      const { readFileSync } = require("fs");
      stderr = readFileSync("/tmp/bullpen-stderr.txt", "utf-8").trim();
    } catch {}

    return {
      stdout,
      stderr,
      success: true,
      data: parseBullpenOutput(stdout),
      error: null,
    };
  } catch (err: any) {
    let stderr = "";
    try {
      const { readFileSync } = require("fs");
      stderr = readFileSync("/tmp/bullpen-stderr.txt", "utf-8").trim();
    } catch {}

    const errorMsg = stderr
      ? parseBullpenError(stderr)
      : err.message || "Unknown error";

    return {
      stdout: err.stdout?.toString() || "",
      stderr,
      success: false,
      data: null,
      error: errorMsg,
    };
  }
}

// High-level commands
export function getTraderActivity(address: string, limit = 10): any[] {
  const result = bullpenExec(
    `polymarket activity --address ${address} --type trade --limit ${limit} --output json`
  );
  return Array.isArray(result.data) ? result.data : [];
}

export function getPositions(): any[] {
  const result = bullpenExec("polymarket positions --output json");
  if (!result.success) {
    console.error(`[executor] positions failed: ${result.error}`);
    return [];
  }
  const data = result.data;
  return data?.positions || (Array.isArray(data) ? data : []);
}

export function buyMarket(slug: string, outcome: string, amount: number): BullpenResult {
  return bullpenExec(
    `polymarket buy ${slug} "${outcome}" ${amount} --yes --output json`
  );
}

export function sellMarket(slug: string, outcome: string, shares: number): BullpenResult {
  return bullpenExec(
    `polymarket sell ${slug} "${outcome}" ${shares} --yes --output json`
  );
}

export function getPrice(slug: string): Map<string, number> {
  const prices = new Map<string, number>();
  const result = bullpenExec(`polymarket price ${slug} --output json`);
  if (!result.success || !result.data) return prices;
  const outcomes = result.data.outcomes || [];
  for (const o of outcomes) {
    prices.set(o.outcome, parseFloat(o.midpoint || o.last_trade || "0"));
  }
  return prices;
}

export function redeemResolved(): string | null {
  const result = bullpenExec("polymarket redeem --yes --output json");
  if (result.success && result.stdout && !result.stdout.includes("nothing")) {
    return result.stdout;
  }
  return null;
}

// src/filters.ts
import { BotConfig } from "./config";

export interface TradeSignal {
  traderName: string;
  traderAddress: string;
  side: "BUY" | "SELL";
  slug: string;
  outcome: string;
  price: number;
  traderAmount: number;
  timestamp: string;
}

export interface FilterState {
  marketExposure: Map<string, number>;  // "slug:outcome" -> total USD exposed
  dailyExposure: number;               // USD committed today
  seenPositions: Set<string>;           // "trader:slug:outcome" — positions trader already holds
  recentSignals: TradeSignal[];         // Recent signals for multi-wallet confirmation
}

export interface FilterResult {
  pass: boolean;
  reason?: string;
}

export function shouldCopyTrade(
  signal: TradeSignal,
  config: BotConfig,
  state: FilterState
): FilterResult {
  // SELL signals always pass — we want to exit when traders exit
  if (signal.side === "SELL") {
    return { pass: true };
  }

  const { filters, risk } = config;

  // Price range filter
  if (signal.price < filters.minPrice) {
    return { pass: false, reason: `price ${signal.price} below min ${filters.minPrice}` };
  }
  if (signal.price > filters.maxPrice) {
    return { pass: false, reason: `price ${signal.price} above max ${filters.maxPrice}` };
  }

  // Minimum trader amount filter
  if (signal.traderAmount < filters.minTraderAmount) {
    return { pass: false, reason: `trader amount $${signal.traderAmount} below min $${filters.minTraderAmount}` };
  }

  // New positions only
  if (filters.newPositionsOnly) {
    const posKey = `${signal.traderName}:${signal.slug}:${signal.outcome}`;
    if (state.seenPositions.has(posKey)) {
      return { pass: false, reason: `${signal.traderName} already holds ${signal.slug}:${signal.outcome}` };
    }
  }

  // Per-market exposure cap
  const marketKey = `${signal.slug}:${signal.outcome}`;
  const currentExposure = state.marketExposure.get(marketKey) || 0;
  if (currentExposure + risk.tradeAmountUsd > risk.maxPerMarket) {
    return { pass: false, reason: `market cap: ${marketKey} at $${currentExposure}/$${risk.maxPerMarket}` };
  }

  // Daily exposure cap
  if (state.dailyExposure + risk.tradeAmountUsd > risk.maxDailyExposure) {
    return { pass: false, reason: `daily cap: $${state.dailyExposure}/$${risk.maxDailyExposure}` };
  }

  // Multi-wallet confirmation (if enabled)
  if (filters.requireMultiWallet) {
    const windowMs = filters.multiWalletWindow * 60 * 1000;
    const cutoff = Date.now() - windowMs;
    const confirming = state.recentSignals.filter(
      (s) =>
        s.slug === signal.slug &&
        s.outcome === signal.outcome &&
        s.traderAddress !== signal.traderAddress &&
        new Date(s.timestamp).getTime() > cutoff
    );
    if (confirming.length === 0) {
      return { pass: false, reason: `multi-wallet: no confirmation for ${signal.slug}:${signal.outcome}` };
    }
  }

  return { pass: true };
}

import Database from "better-sqlite3";
import { join } from "path";

export function createDb(dataDir: string): Database.Database {
  const db = new Database(join(dataDir, "copybot.db"));
  db.pragma("journal_mode = WAL");

  db.exec(`
    CREATE TABLE IF NOT EXISTS trades (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      timestamp TEXT NOT NULL,
      trader TEXT NOT NULL,
      trader_address TEXT NOT NULL,
      action TEXT NOT NULL,
      market TEXT,
      slug TEXT NOT NULL,
      outcome TEXT NOT NULL,
      trader_amount REAL,
      our_amount REAL,
      entry_price REAL,
      paper_shares REAL,
      status TEXT NOT NULL,
      error TEXT,
      result TEXT,
      is_real INTEGER NOT NULL DEFAULT 0
    );

    CREATE TABLE IF NOT EXISTS daily_stats (
      date TEXT PRIMARY KEY,
      exposure REAL DEFAULT 0,
      trades_count INTEGER DEFAULT 0,
      pnl_realized REAL DEFAULT 0
    );

    CREATE INDEX IF NOT EXISTS idx_trades_slug ON trades(slug, outcome);
    CREATE INDEX IF NOT EXISTS idx_trades_status ON trades(status);
    CREATE INDEX IF NOT EXISTS idx_trades_timestamp ON trades(timestamp);
  `);

  return db;
}

export function insertTrade(db: Database.Database, trade: {

  trader: string;
  traderAddress: string;
  action: string;
  market: string;
  slug: string;
  outcome: string;
  traderAmount: number;
  ourAmount: number;
  entryPrice: number;
  paperShares: number;
  status: string;
  error?: string;
  result?: string;
  isReal: boolean;

  db.prepare(`
    INSERT INTO trades (timestamp, trader, trader_address, action, market, slug, outcome,
      trader_amount, our_amount, entry_price, paper_shares, status, error, result, is_real)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `).run(
    trade.timestamp, trade.trader, trade.traderAddress, trade.action,
    trade.market, trade.slug, trade.outcome, trade.traderAmount,
    trade.ourAmount, trade.entryPrice, trade.paperShares, trade.status,
    trade.error || null, trade.result || null, trade.isReal ? 1 : 0
  );
}

  let sql = "SELECT * FROM trades WHERE 1=1";
  const params: any[] = [];

  if (opts?.status) {
    sql += " AND status = ?";
    params.push(opts.status);
  }
  if (opts?.isReal !== undefined) {
    sql += " AND is_real = ?";
    params.push(opts.isReal ? 1 : 0);
  }

  sql += " ORDER BY id DESC";

  if (opts?.limit) {
    sql += " LIMIT ?";
    params.push(opts.limit);
  }

  return db.prepare(sql).all(...params);
}

export function getMarketExposure(db: Database.Database): Map<string, number> {
  const rows = db.prepare(`
    SELECT slug || ':' || outcome AS key, SUM(our_amount) AS exposure
    FROM trades
    WHERE action = 'BUY' AND status IN ('success', 'failed')
    GROUP BY slug, outcome
  `).all() as any[];

  const map = new Map<string, number>();
  for (const row of rows) {
    map.set(row.key, row.exposure);
  }
  return map;
}

export function getDailyExposure(db: Database.Database): number {
  const today = new Date().toISOString().split("T")[0];
  const row = db.prepare(`
    SELECT COALESCE(SUM(our_amount), 0) AS total
    FROM trades
    WHERE action = 'BUY' AND status = 'success' AND timestamp >= ?
  `).get(today + "T00:00:00.000Z") as any;
  return row?.total || 0;
}

// src/tracker.ts

export interface PnlResult {
  pnl: number;
  invested: number;
  returnPct: number;
  positions: Array<{
    slug: string;
    outcome: string;
    entry: number;
    shares: number;
    current: number;
    pnl: number;
    trader?: string;
  }>;
}

export function computePaperPnl(
  trades: any[],
  prices: Map<string, Map<string, number>>
): PnlResult {
  return computePnlFromTrades(trades, prices, false);
}

export function computeRealPnl(
  trades: any[],
  prices: Map<string, Map<string, number>>
): PnlResult {
  const realTrades = trades.filter((t) => t.is_real === 1 && t.status === "success");
  return computePnlFromTrades(realTrades, prices, true);
}

function computePnlFromTrades(
  trades: any[],
  prices: Map<string, Map<string, number>>,
  realOnly: boolean
): PnlResult {
  const buys = trades.filter((t) => t.action === "BUY" && t.entry_price > 0);

  let totalPnl = 0;
  let totalInvested = 0;
  const positions: PnlResult["positions"] = [];

  for (const t of buys) {
    const slugPrices = prices.get(t.slug);
    const currentPrice = slugPrices?.get(t.outcome) || 0;
    if (currentPrice === 0) continue;

    const positionPnl = (currentPrice - t.entry_price) * t.paper_shares;
    totalPnl += positionPnl;
    totalInvested += t.our_amount;

    positions.push({
      slug: t.slug,
      outcome: t.outcome,
      entry: t.entry_price,
      shares: t.paper_shares,
      current: currentPrice,
      pnl: positionPnl,
      trader: t.trader,
    });
  }

  return {
    pnl: Math.round(totalPnl * 100) / 100,
    invested: totalInvested,
    returnPct: totalInvested > 0 ? Math.round((totalPnl / totalInvested) * 10000) / 100 : 0,
    positions,
  };
}

interface ParsedTrade {
  id: string;
  timestamp: string;
  side: "BUY" | "SELL";
  market: string;
  slug: string;
  outcome: string;
  amount: number;
  price: number;
}

function parseActivity(raw: any[], traderAddress: string): ParsedTrade[] {
  return raw.map((t) => ({
    id: t.transaction_hash || `${t.timestamp}-${t.slug}-${t.outcome}`,
    timestamp: t.timestamp || "",
    side: (t.side || "").toUpperCase() as "BUY" | "SELL",
    market: t.title || "",
    slug: t.slug || "",
    outcome: t.outcome || "",
    amount: t.usdc_size || t.size || 0,

}

function processTrader(trader: TraderConfig): void {
  const rawActivity = getTraderActivity(trader.address);
  const trades = parseActivity(rawActivity, trader.address);

  for (const trade of trades) {
    if (seenTrades.has(trade.id)) continue;
    seenTrades.add(trade.id);

    if (isFirstRun) {
      // Track initial positions but don't copy
      if (trade.side === "BUY") {
        seenPositions.add(`${trader.name}:${trade.slug}:${trade.outcome}`);
      }
      continue;
    }

    // Build signal
    const signal: TradeSignal = {
      traderName: trader.name,
      traderAddress: trader.address,
      side: trade.side,
      slug: trade.slug,
      outcome: trade.outcome,
      price: trade.price,
      traderAmount: trade.amount,
      timestamp: trade.timestamp,
    };

    // Track for multi-wallet confirmation
    recentSignals.push(signal);
    // Trim old signals (keep last 30 minutes)
    const cutoff = Date.now() - 30 * 60 * 1000;
    recentSignals = recentSignals.filter(
      (s) => new Date(s.timestamp).getTime() > cutoff
    );

    // Apply filters
    const filterState: FilterState = {
      marketExposure: getMarketExposure(db),
      dailyExposure: getDailyExposure(db),
      seenPositions,
      recentSignals,
    };

    const filterResult = shouldCopyTrade(signal, config, filterState);

    if (!filterResult.pass) {
      console.log(`  [SKIP] ${trader.name} ${trade.side} $${trade.amount.toFixed(0)} ${trade.slug} → ${filterResult.reason}`);
      insertTrade(db, {
        timestamp: new Date().toISOString(),
        trader: trader.name,
        traderAddress: trader.address,
        action: trade.side,
        market: trade.market,
        slug: trade.slug,
        outcome: trade.outcome,
        traderAmount: trade.amount,
        ourAmount: 0,
        entryPrice: trade.price,
        paperShares: 0,
        status: "skipped",
        error: filterResult.reason,
        isReal: false,
      });
      continue;
    }

    // Execute trade
    console.log(`  [NEW] ${trader.name} ${trade.side} $${trade.amount.toFixed(0)} on ${trade.slug} → ${trade.outcome} @ ${trade.price}`);

    if (trade.side === "BUY") {
      executeBuySignal(trader, trade);
    } else {
      executeSellSignal(trader, trade);
    }
  }
}

function executeBuySignal(trader: TraderConfig, trade: ParsedTrade): void {
  const entryPrice = trade.price;
  const paperShares = entryPrice > 0 ? config.risk.tradeAmountUsd / entryPrice : 0;

  if (!config.paperMode) {
    const result = buyMarket(trade.slug, trade.outcome, config.risk.tradeAmountUsd);
    if (result.success) {
      console.log(`  [BUY] Executed: $${config.risk.tradeAmountUsd} on ${trade.slug} ${trade.outcome}`);
      insertTrade(db, {
        timestamp: new Date().toISOString(),
        trader: trader.name, traderAddress: trader.address,
        action: "BUY", market: trade.market, slug: trade.slug, outcome: trade.outcome,
        traderAmount: trade.amount, ourAmount: config.risk.tradeAmountUsd,
        entryPrice, paperShares, status: "success", result: result.stdout, isReal: true,
      });
      seenPositions.add(`${trader.name}:${trade.slug}:${trade.outcome}`);
      return;
    }
    console.error(`  [FAIL] Buy failed: ${result.error}`);
  }

  // Paper trade (either paper mode or real trade failed)
  console.log(`  [PAPER] $${config.risk.tradeAmountUsd} on ${trade.slug} ${trade.outcome} @ ${entryPrice.toFixed(2)}`);
  insertTrade(db, {
    timestamp: new Date().toISOString(),
    trader: trader.name, traderAddress: trader.address,
    action: "BUY", market: trade.market, slug: trade.slug, outcome: trade.outcome,
    traderAmount: trade.amount, ourAmount: config.risk.tradeAmountUsd,
    entryPrice, paperShares, status: config.paperMode ? "paper" : "failed",
    isReal: false,
  });
  seenPositions.add(`${trader.name}:${trade.slug}:${trade.outcome}`);
}

function executeSellSignal(trader: TraderConfig, trade: ParsedTrade): void {
  if (!config.paperMode) {
    // Check if we hold this position
    const positions = getPositions();
    const pos = positions.find(
      (p: any) => (p.slug || p.market_slug) === trade.slug && p.outcome === trade.outcome
    );
    if (!pos) {
      console.log(`  [SKIP] Sell signal but we don't hold ${trade.slug} ${trade.outcome}`);
      return;
    }
    const shares = parseFloat(pos.size || pos.shares || "0");
    const sellShares = Math.max(1, Math.floor(shares * 0.5)); // Sell 50% on any sell signal
    const result = sellMarket(trade.slug, trade.outcome, sellShares);
    if (result.success) {
      console.log(`  [SELL] Sold ${sellShares} shares of ${trade.slug} ${trade.outcome}`);
      insertTrade(db, {
        timestamp: new Date().toISOString(),
        trader: trader.name, traderAddress: trader.address,
        action: "SELL", market: trade.market, slug: trade.slug, outcome: trade.outcome,
        traderAmount: trade.amount, ourAmount: sellShares,
        entryPrice: 0, paperShares: 0, status: "success", result: result.stdout, isReal: true,
      });
      return;
    }
    console.error(`  [FAIL] Sell failed: ${result.error}`);
  }

  // Paper sell
  insertTrade(db, {
    timestamp: new Date().toISOString(),
    trader: trader.name, traderAddress: trader.address,
    action: "SELL", market: trade.market, slug: trade.slug, outcome: trade.outcome,
    traderAmount: trade.amount, ourAmount: 0,
    entryPrice: 0, paperShares: 0, status: config.paperMode ? "paper" : "failed",
    isReal: false,
  });
  // Remove from seen positions so we can re-enter
  seenPositions.delete(`${trader.name}:${trade.slug}:${trade.outcome}`);
}

function refreshPnl(): void {
  try {
    const allTrades = getTrades(db);
    // Get unique slugs
    const slugs = [...new Set(allTrades.filter((t: any) => t.action === "BUY").map((t: any) => t.slug))];
    const prices = new Map<string, Map<string, number>>();
    for (const slug of slugs.slice(0, 50)) { // Limit to 50 to avoid rate limits
      prices.set(slug, getPrice(slug));
    }

    const paper = computePaperPnl(allTrades, prices);
    const real = computeRealPnl(allTrades, prices);

    writeFileSync(PAPER_PNL_FILE, JSON.stringify({
      paper,
      real,
      updatedAt: new Date().toISOString(),
    }));
    console.log(`  [P&L] Paper: $${paper.pnl} (${paper.returnPct}%) | Real: $${real.pnl} (${real.returnPct}%)`);
  } catch (err: any) {
    console.error(`  [!] P&L refresh failed: ${err.message}`);
  }
}

  if (req.url === "/" || req.url === "/index.html") {
    res.writeHead(200, { "Content-Type": "text/html" });
    res.end(HTML);
    return;
  }

  res.setHeader("Content-Type", "application/json");
  res.setHeader("Access-Control-Allow-Origin", "*");

  if (req.url === "/api/data") {
    const db = getDb();
    let trades: any[] = [];
    if (db) {
      trades = db.prepare("SELECT * FROM trades ORDER BY id DESC LIMIT 200").all();
      db.close();
    }
    const status = getStatus();
    const pnl = getPnl();
    res.writeHead(200);
    res.end(JSON.stringify({ running: status.running, trades, pnl }));
  } else if (req.url === "/api/start" && req.method === "POST") {
    const { writeFileSync } = require("fs");
    writeFileSync(STATUS_FILE, JSON.stringify({ running: true }));
    res.writeHead(200);
    res.end(JSON.stringify({ running: true }));
  } else if (req.url === "/api/stop" && req.method === "POST") {
    const { writeFileSync } = require("fs");
    writeFileSync(STATUS_FILE, JSON.stringify({ running: false }));
    res.writeHead(200);
    res.end(JSON.stringify({ running: false }));
  } else {
    res.writeHead(404);
    res.end(JSON.stringify({ error: "not found" }));
  }
});
